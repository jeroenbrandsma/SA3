		\newcommand{\getNext}[1]{%
			%check if counter with that name excists
			\@ifundefined{c@#1}
			{% the counter doesn't exist
				\newcounter{#1}\setcounter{#1}{1}%
			}
			{% the counter exists
				\stepcounter{#1}%
			}%
		}
						
		\newcommand{\getNr}[1]{%
			\getNext{#1}
			\arabic{#1}
		}
						
		\newcommand{\nextNrRef}[1]{%
			\getNext{#1}\manuallabel{#1:\arabic{#1}}{\uppercase{#1}-\arabic{#1}}\arabic{#1}
		}
